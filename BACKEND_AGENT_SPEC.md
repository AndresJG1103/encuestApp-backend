# BACKEND AGENT SPECIFICATION — LearnPulse
> **Audience:** AI coding agent operating inside an IDE (Cursor / Claude Code / Copilot).  
> **Tone:** Deterministic. No ambiguity. Every decision is pre-made. Follow exactly.  
> **Stack:** NestJS 10 · TypeScript 5 · Prisma 5 · PostgreSQL 16 · Redis 7 · BullMQ 5 · AWS S3 · Docker

---

## 0. GROUND RULES FOR THE AGENT

1. **Never use `any`** — TypeScript strict mode is ON (`"strict": true` in tsconfig).
2. **Prisma is the only ORM** — no TypeORM, no raw `pg`, no Knex. All DB access goes through `PrismaService`.
3. **Hexagonal architecture is mandatory** — every module has `domain/`, `application/`, `infrastructure/`, `presentation/` layers. Details in §3.
4. **No business logic in Controllers** — Controllers only validate DTOs, call Use Cases, and return responses.
5. **No business logic in Prisma repositories** — repositories only translate domain calls to Prisma queries.
6. **All secrets from `ConfigService`** — never `process.env` directly outside `config/` folder.
7. **Every endpoint is protected by `JwtAuthGuard` globally**, unless decorated with `@Public()`.
8. **Multi-tenancy is enforced at repository level** — every query that touches tenant-scoped data MUST include `WHERE tenantId = X`. The `tenantId` comes from the JWT payload, extracted by `@CurrentTenant()` decorator.
9. **UUIDs everywhere** — all PKs are `uuid()` generated by Prisma default.
10. **Soft deletes** — use `deletedAt TIMESTAMP NULL` + `@ignore` filtered queries. Never hard delete tenant data.

```typescript
import { Reflector } from '@nestjs/core';
import { AuthGuard } from '@nestjs/passport';

export class JwtAuthGuard extends AuthGuard('jwt') {
  constructor(private reflector: Reflector) { super(); }

  canActivate(context: ExecutionContext) {
    const isPublic = this.reflector.getAllAndOverride<boolean>(IS_PUBLIC_KEY, [

      context.getClass(),
```

### 4.5 RolesGuard

```typescript
// src/shared/guards/roles.guard.ts
import { CanActivate, ExecutionContext, ForbiddenException, Injectable } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { ROLES_KEY } from '../decorators/roles.decorator';
import { JwtPayload } from '../types/jwt-payload.type';

@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredRoles = this.reflector.getAllAndOverride<string[]>(ROLES_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);
    if (!requiredRoles || requiredRoles.length === 0) return true;

    const { user }: { user: JwtPayload } = context.switchToHttp().getRequest();
    const hasRole = requiredRoles.some(role => user.roles.includes(role));
    if (!hasRole) throw new ForbiddenException('Insufficient role');
    return true;
  }
}
```

```typescript
// src/shared/interceptors/transform.interceptor.ts
import { CallHandler, ExecutionContext, Injectable, NestInterceptor } from '@nestjs/common';
import { Observable, map } from 'rxjs';

export interface ApiResponse<T> {
  success: boolean;
  data: T;
  meta?: Record<string, unknown>;
}

@Injectable()
export class TransformInterceptor<T> implements NestInterceptor<T, ApiResponse<T>> {
  intercept(_context: ExecutionContext, next: CallHandler): Observable<ApiResponse<T>> {
    return next.handle().pipe(
      map(data => ({
        success: true,
        data: data?.data ?? data,
        meta: data?.meta,
      })),
    );
  }
}
```

```typescript
// src/shared/filters/http-exception.filter.ts
import { ArgumentsHost, Catch, ExceptionFilter, HttpException, HttpStatus } from '@nestjs/common';
import { Response } from 'express';

@Catch()
export class GlobalExceptionFilter implements ExceptionFilter {
  catch(exception: unknown, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse<Response>();

    const status = exception instanceof HttpException
      ? exception.getStatus()
      : HttpStatus.INTERNAL_SERVER_ERROR;

    const message = exception instanceof HttpException
      ? exception.getResponse()
      : 'Internal server error';

    response.status(status).json({
      success: false,
      error: message,
      timestamp: new Date().toISOString(),
    });
  }
}
```

```typescript
import { Injectable, UnauthorizedException } from '@nestjs/common';
import { PassportStrategy } from '@nestjs/passport';
import { ExtractJwt, Strategy } from 'passport-jwt';
import { ConfigService } from '@nestjs/config';
import { JwtPayload } from '@shared/types/jwt-payload.type';
import { RedisService } from '@shared/redis/redis.service';

@Injectable()
export class JwtStrategy extends PassportStrategy(Strategy) {
  constructor(
    private config: ConfigService,
    private redis: RedisService,
  ) {
    super({
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      secretOrKey: config.get<string>('JWT_SECRET'),
      passReqToCallback: false,
    });
  }

  async validate(payload: JwtPayload): Promise<JwtPayload> {
    // Check blacklist
    const isBlacklisted = await this.redis.get(`bl:${payload.jti}`);
    if (isBlacklisted) throw new UnauthorizedException('Token revoked');
    return payload;
  }
}
```

### 5.2 Login Use Case

```typescript
// src/modules/auth/application/use-cases/login.use-case.ts
import { Injectable, UnauthorizedException } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { ConfigService } from '@nestjs/config';
import { PrismaService } from '@shared/prisma/prisma.service';
import { RedisService } from '@shared/redis/redis.service';
import * as bcrypt from 'bcrypt';
import { v4 as uuidv4 } from 'uuid';

export interface LoginDto {
  email: string;
  password: string;
  tenantSlug: string;
}

export interface AuthTokens {
  accessToken: string;
  refreshToken: string;
}

@Injectable()
export class LoginUseCase {
  constructor(
    private prisma: PrismaService,
    private jwt: JwtService,
    private config: ConfigService,
    private redis: RedisService,
  ) {}

  async execute(dto: LoginDto): Promise<AuthTokens> {
    // 1. Find tenant by slug
    const tenant = await this.prisma.tenant.findUnique({
      where: { slug: dto.tenantSlug, deletedAt: null },
    });
    if (!tenant) throw new UnauthorizedException('Tenant not found');

    // 2. Find user within tenant
    const user = await this.prisma.user.findUnique({
      where: { email_tenantId: { email: dto.email, tenantId: tenant.id } },
      include: { roles: true },
    });
    if (!user || !user.isActive) throw new UnauthorizedException('Invalid credentials');

    // 3. Compare password
    const valid = await bcrypt.compare(dto.password, user.passwordHash);
    if (!valid) throw new UnauthorizedException('Invalid credentials');

    // 4. Build payload
    const jti = uuidv4();
    const payload = {
      sub: user.id,
      tenantId: tenant.id,
      roles: user.roles.map(r => r.role),
      jti,
    };

    // 5. Sign tokens
    const accessToken = this.jwt.sign(payload, {
      secret: this.config.get('JWT_SECRET'),
      expiresIn: '15m',
    });

    const refreshToken = this.jwt.sign(
      { sub: user.id, tenantId: tenant.id, jti },
      { secret: this.config.get('JWT_REFRESH_SECRET'), expiresIn: '7d' },
    );

    // 6. Store refresh token in Redis
    const rtKey = `rt:${user.id}:${jti}`;
    await this.redis.set(rtKey, refreshToken, 60 * 60 * 24 * 7); // 7 days TTL

    return { accessToken, refreshToken };
  }
}
```

```typescript
import { Injectable } from '@nestjs/common';
import { PrismaService } from '@shared/prisma/prisma.service';
import { IFormRepository, FindFormsFilter } from '../domain/repositories/form.repository.interface';
import { Form } from '@prisma/client';
import { PaginatedResult, PaginationParams } from '@shared/types/paginated.type';

@Injectable()
export class FormPrismaRepository implements IFormRepository {
  constructor(private prisma: PrismaService) {}

  async findById(id: string, tenantId: string): Promise<Form | null> {
    return this.prisma.form.findFirst({
      where: { id, tenantId, deletedAt: null },
      include: {
        sections: {
          orderBy: { order: 'asc' },
          include: { items: { orderBy: { order: 'asc' } } },
        },
        createdBy: { select: { id: true, firstName: true, lastName: true } },
      },
    });
  }

  async findMany(filter: FindFormsFilter, pagination: PaginationParams): Promise<PaginatedResult<Form>> {
    const page = pagination.page ?? 1;
    const limit = pagination.limit ?? 20;
    const skip = (page - 1) * limit;

    const where = {
      tenantId: filter.tenantId,
      deletedAt: null,
      ...(filter.type && { type: filter.type as any }),
      ...(filter.status && { status: filter.status as any }),
      ...(filter.createdById && { createdById: filter.createdById }),
    };

    const [data, total] = await this.prisma.$transaction([
      this.prisma.form.findMany({ where, skip, take: limit, orderBy: { createdAt: 'desc' } }),
      this.prisma.form.count({ where }),
    ]);

    return { data, meta: { total, page, limit, totalPages: Math.ceil(total / limit) } };
  }

  async create(data: Partial<Form>): Promise<Form> {
    return this.prisma.form.create({ data: data as any });
  }

  async update(id: string, tenantId: string, data: Partial<Form>): Promise<Form> {
    return this.prisma.form.update({ where: { id }, data: { ...data, updatedAt: new Date() } });
  }

  async softDelete(id: string, _tenantId: string): Promise<void> {
    await this.prisma.form.update({ where: { id }, data: { deletedAt: new Date() } });
  }

  async incrementVersion(id: string, _tenantId: string): Promise<Form> {
    return this.prisma.form.update({ where: { id }, data: { version: { increment: 1 } } });
  }
}
```

```typescript
import { ConflictException, Injectable, NotFoundException } from '@nestjs/common';
import { PrismaService } from '@shared/prisma/prisma.service';
import { RedisService } from '@shared/redis/redis.service';

export interface StartSessionDto {
  formId: string;
  userId: string;
  tenantId: string;
}

@Injectable()
export class StartSessionUseCase {
  constructor(
    private prisma: PrismaService,
    private redis: RedisService,
  ) {}

  async execute(dto: StartSessionDto) {
    // 1. Validate form exists and is published in this tenant
    const form = await this.prisma.form.findFirst({
      where: { id: dto.formId, tenantId: dto.tenantId, status: 'PUBLISHED', deletedAt: null },
    });
    if (!form) throw new NotFoundException('Form not found or not published');

    const config = form.config as any;

    // 2. Check existing in-progress session
    const existing = await this.prisma.responseSession.findFirst({
      where: { formId: dto.formId, userId: dto.userId, status: 'IN_PROGRESS' },
    });
    if (existing) return existing; // Resume existing session

    // 3. Check maxAttempts if configured
    if (config.maxAttempts) {
      const completedCount = await this.prisma.responseSession.count({
        where: { formId: dto.formId, userId: dto.userId, status: 'COMPLETED' },
      });
      if (completedCount >= config.maxAttempts) {
        throw new ConflictException('Maximum attempts reached');
      }
    }

    // 4. Get attempt number
    const attemptCount = await this.prisma.responseSession.count({
      where: { formId: dto.formId, userId: dto.userId },
    });

    // 5. Create session
    const session = await this.prisma.responseSession.create({
      data: {
        formId: dto.formId,
        userId: dto.userId,
        attempt: attemptCount + 1,
        status: 'IN_PROGRESS',
        progressSnapshot: { currentSectionIndex: 0, visitedSectionIds: [], answeredItemIds: [], lastSavedAt: new Date().toISOString() },
      },
    });

    // 6. Update assignment status
    await this.prisma.assignment.updateMany({
      where: { formId: dto.formId, userId: dto.userId, status: 'PENDING' },
      data: { status: 'IN_PROGRESS' },
    });

    // 7. Increment Redis analytics counter
    await this.redis.incr(`stats:${dto.formId}:starts`);

    return session;
  }
}
```

### 8.2 SubmitAnswer Use Case — CORE LOGIC

```typescript
// src/modules/response/application/use-cases/submit-answer.use-case.ts
import { BadRequestException, Injectable, NotFoundException } from '@nestjs/common';
import { PrismaService } from '@shared/prisma/prisma.service';
import { RedisService } from '@shared/redis/redis.service';

export interface SubmitAnswerDto {
  sessionId: string;
  itemId: string;
  answer: Record<string, unknown>;   // { selected: string[] } | { text: string } | { value: number|boolean }
  timeSpentMs: number;
  userId: string;
}

@Injectable()
export class SubmitAnswerUseCase {
  constructor(
    private prisma: PrismaService,
    private redis: RedisService,
  ) {}

  async execute(dto: SubmitAnswerDto) {
    // 1. Validate session belongs to user and is in progress
    const session = await this.prisma.responseSession.findFirst({
      where: { id: dto.sessionId, userId: dto.userId, status: 'IN_PROGRESS' },
    });
    if (!session) throw new NotFoundException('Active session not found');

    // 2. Fetch item — this is where correctAnswers live
    const item = await this.prisma.item.findUnique({ where: { id: dto.itemId } });
    if (!item) throw new NotFoundException('Item not found');
    if (item.type !== 'QUESTION') throw new BadRequestException('Item is not a question');

    const content = item.content as any;

    // 3. Evaluate correctness
    // correctAnswers is undefined/null for opinion-only questions (surveys)
    let isCorrect: boolean | null = null;
    if (content.correctAnswers && Array.isArray(content.correctAnswers)) {
      isCorrect = this.evaluate(content, dto.answer);
    }

    // 4. Upsert item response (allow resubmission within same session)
    const itemResponse = await this.prisma.itemResponse.upsert({
      where: { sessionId_itemId: { sessionId: dto.sessionId, itemId: dto.itemId } },
      create: {
        sessionId: dto.sessionId,
        itemId: dto.itemId,
        answer: dto.answer,
        isCorrect,
        timeSpentMs: dto.timeSpentMs,
      },
      update: {
        answer: dto.answer,
        isCorrect,
        timeSpentMs: dto.timeSpentMs,
        answeredAt: new Date(),
      },
    });

    // 5. Update progressSnapshot in Redis (fast cache) and DB
    const snapshot = (session.progressSnapshot as any) ?? {};
    const answeredItemIds: string[] = snapshot.answeredItemIds ?? [];
    if (!answeredItemIds.includes(dto.itemId)) {
      answeredItemIds.push(dto.itemId);
    }
    const newSnapshot = { ...snapshot, answeredItemIds, lastSavedAt: new Date().toISOString() };

    await this.prisma.responseSession.update({
      where: { id: dto.sessionId },
      data: { progressSnapshot: newSnapshot },
    });
    await this.redis.set(`sess:${dto.sessionId}`, JSON.stringify(newSnapshot), 48 * 3600);

    return itemResponse;
  }

  private evaluate(content: any, answer: any): boolean {
    const correct: string[] = content.correctAnswers;

    switch (content.questionType) {
      case 'SINGLE_CHOICE':
        return correct.includes(answer.selected?.[0]);

      case 'MULTIPLE_CHOICE':
        const selected: string[] = answer.selected ?? [];
        return (
          selected.length === correct.length &&
          correct.every(c => selected.includes(c))
        );

      case 'BOOLEAN':
        return String(answer.value) === String(correct[0]);

      case 'SCALE':
      case 'RATING':
        return String(answer.value) === String(correct[0]);

      default:
        return false; // OPEN_TEXT — no auto-evaluation
    }
  }
}
```

### 8.3 CompleteSession Use Case

```typescript
// src/modules/response/application/use-cases/complete-session.use-case.ts
import { BadRequestException, Injectable, NotFoundException } from '@nestjs/common';
import { PrismaService } from '@shared/prisma/prisma.service';
import { RedisService } from '@shared/redis/redis.service';
import { InjectQueue } from '@nestjs/bull';
import { Queue } from 'bull';

@Injectable()
export class CompleteSessionUseCase {
  constructor(
    private prisma: PrismaService,
    private redis: RedisService,
    @InjectQueue('certificate-queue') private certQueue: Queue,
    @InjectQueue('analytics-queue') private analyticsQueue: Queue,
  ) {}

  async execute(sessionId: string, userId: string) {
    const session = await this.prisma.responseSession.findFirst({
      where: { id: sessionId, userId, status: 'IN_PROGRESS' },
      include: {
        form: true,
        itemResponses: { include: { item: true } },
      },
    });
    if (!session) throw new NotFoundException('Active session not found');

    const config = session.form.config as any;

    // 1. Calculate score (only for scored items)
    const scoredResponses = session.itemResponses.filter(r => r.isCorrect !== null);
    let score: number | null = null;
    let passed: boolean | null = null;

    if (scoredResponses.length > 0) {
      const totalPoints = scoredResponses.reduce((sum, r) => {
        const content = r.item.content as any;
        return sum + (content.points ?? 1);
      }, 0);

      const earnedPoints = scoredResponses
        .filter(r => r.isCorrect)
        .reduce((sum, r) => {
          const content = r.item.content as any;
          return sum + (content.points ?? 1);
        }, 0);

      score = totalPoints > 0 ? (earnedPoints / totalPoints) * 100 : 0;
      passed = config.passingScore != null ? score >= config.passingScore : null;
    }

    // 2. Update session
    const completed = await this.prisma.responseSession.update({
      where: { id: sessionId },
      data: {
        status: 'COMPLETED',
        completedAt: new Date(),
        score: score != null ? score : undefined,
        passed: passed ?? undefined,
      },
    });

    // 3. Update assignment
    await this.prisma.assignment.updateMany({
      where: { formId: session.formId, userId, status: 'IN_PROGRESS' },
      data: { status: 'COMPLETED' },
    });

    // 4. Queue certificate generation if passed
    if (passed === true && config.certificateOnPass) {
      await this.certQueue.add('generate', { sessionId, userId, formId: session.formId });
    }

    // 5. Queue analytics update
    await this.analyticsQueue.add('session-completed', { sessionId, formId: session.formId, score, passed });

    // 6. Increment Redis counter
    await this.redis.incr(`stats:${session.formId}:completes`);

    // 7. Clean session cache
    await this.redis.del(`sess:${sessionId}`);

    return { session: completed, score, passed };
  }
}
```

```typescript
export function serializeItem(item: Item, userRoles: RoleType[]): Item {

    !userRoles.includes(RoleType.CREATOR) &&
    !userRoles.includes(RoleType.TENANT_ADMIN) &&
    !userRoles.includes(RoleType.REVIEWER);
```

```typescript
      Bucket: this.config.get('AWS_S3_BUCKET'),
      Key: fileKey,
      ContentType: dto.mimeType,
```
